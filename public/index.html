Engine template

(ns car.engine
  (:require [car.core :refer [set-engine!]]))

(set-engine! (fn [] :gas))
(set-engine! (fn [] :break))
(set-engine! (fn [] :none))

(ns car.wheel
  (:require [car.core :refer [set-wheel!]]))

(set-wheel! (fn [] :left))
(set-wheel! (fn [] :right))
(set-wheel! (fn [] :none))
